$('.pedir-permissao').click(function(event) {
    // usuário ainda não especificou se aprova notificações desse site
    // então perguntamos
    if (Notification.permissionLevel() === 'default') {
        Notification.requestPermission(mostraNotificacao)
    // já foi dada a permissão
    } else if (Notification.permissionLevel() === 'granted') {
        mostraNotificacao()
    // usuário negou a permissão para mostrarmos notificação
    } else if (Notification.permissionLevel() === 'denied') {
        return false
    }
})

function mostraNotificacao() {
    notificacao = new Notification('Minha primeira notificação')
    notificacao.show()
}
