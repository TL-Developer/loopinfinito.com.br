body {
    color: #333;
    background-color: #DDD;
}

/* ambiente escuro */
@media (luminosity: dim) {
    /* diminui o contraste do texto */
    body {
        color: #666;
        background-color: #AAA;
    }
}

/* ambiente muito claro */
@media (luminosity: washed) {
    /* aumenta o contraste do texto */
    body {
        color: black;
        background-color: white;
    }
}
